word = input('give me a word: ')
reverseWord = ''
print(len(word))
for i in range(len(word) - 1, -1, -1): 
    print(word[i])
    reverseWord += word[i] #reverseWord = reverseWord + wrod[i]

if reverseWord == word:
    print(f'reverseWord: {reverseWord} == word: {word}')
    print('is palindromo')
else:
    print(f'reverseWord: {reverseWord} != word: {word}')
    print('Not is palindromo')
