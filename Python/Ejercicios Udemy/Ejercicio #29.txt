# Ejecicio #29: Escribir un programa que pida al usuario una palabra y muestre por pantalla el n√∫mero de veces que
# contiene cada vocal.
numbersA = 0
numbersE = 0
numbersI = 0
numbersO = 0
numbersU = 0
word = input('give a word: ')
word = word.lower()
for i in word:
    if i == 'a':
        numbersA += 1
    elif i == 'e':
        numbersE += 1
    elif i == 'i':
        numbersI += 1
    elif i == 'o':
        numbersO += 1
    elif i == 'u':
        numbersU += 1

print(f'''
provided word: {word}
there is {numbersA} A's in {word}
there is {numbersE} E's in {word}
there is {numbersI} I's in {word}
there is {numbersO} O's in {word}
there is {numbersU} U's in {word}
''')
