N = min(max(int(input()), 1), 365) #N solo puede estar entre 1 y 365
promedioDias = 0
promedioTotal = 0
acumulador = 0
for i in range(1, N+1): #Pa que i tome todos los valores entre 1 y 6
    promedioDias = float(input())
    promedioTotal = int(promedioDias*i - acumulador)
    acumulador += promedioTotal
    print(promedioTotal)
